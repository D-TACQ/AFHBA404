46 int hba_sfp_get_value(struct AFHBA_DEV *adev, unsigned gpio)
47 {
48  int rn = SFP_I2C_DATA_REG(adev->sfp);
49  u32 reg = afhba_read_reg(adev, rn);
50  int bit = (reg&(1 << gpio)) != 0;
51  dev_dbg(pdev(adev), "read:%d", bit);
52  return bit;
53 }
54 void hba_sfp_set_value(struct AFHBA_DEV *adev, unsigned gpio, int value)
55 {
56  int rn = SFP_I2C_DATA_REG(adev->sfp);
57  u32 reg = afhba_read_reg(adev, rn);
58 
59  dev_dbg(pdev(adev), "write:%d", value);
60  if (value){
61  reg |= 1 << (gpio+8);
62  }else{
63  reg &= ~(1 << (gpio+8));
64  }
65  afhba_write_reg(adev, rn, reg);
66 }

156 static int i2c_hba_sfp_probe(struct platform_device *pdev)
157 {
158  struct i2c_gpio_platform_data *pdata;
159  struct i2c_algo_bit_data *bit_data;
160  struct i2c_adapter *adap;
161  int ret;
162 
163  pdata = pdev->dev.platform_data;
164  if (!pdata)
165  return -ENXIO;
166 
167  ret = -ENOMEM;
168  adap = kzalloc(sizeof(struct i2c_adapter), GFP_KERNEL);
169  if (!adap)
170  goto err_alloc_adap;
171  bit_data = kzalloc(sizeof(struct i2c_algo_bit_data), GFP_KERNEL);
172  if (!bit_data)
173  goto err_alloc_bit_data;
174 
175  ret = hba_sfp_request(GETD(pdev), pdata->sda_pin, "sda");
176  if (ret)
177  goto err_request_sda;
178  ret = hba_sfp_request(GETD(pdev), pdata->scl_pin, "scl");
179  if (ret)
180  goto err_request_scl;
181 
182  if (pdata->sda_is_open_drain) {
183  hba_sfp_direction_output(GETD(pdev), pdata->sda_pin, 1);
184  bit_data->setsda = i2c_hba_sfp_setsda_val;
185  } else {
186  hba_sfp_direction_input(GETD(pdev), pdata->sda_pin);
187  bit_data->setsda = i2c_hba_sfp_setsda_dir;
188  }
189 
190  if (pdata->scl_is_open_drain || pdata->scl_is_output_only) {
191  hba_sfp_direction_output(GETD(pdev), pdata->scl_pin, 1);
192  bit_data->setscl = i2c_hba_sfp_setscl_val;
193  } else {
194  hba_sfp_direction_input(GETD(pdev), pdata->scl_pin);
195  bit_data->setscl = i2c_hba_sfp_setscl_dir;
196  }
197 
198  if (!pdata->scl_is_output_only)
199  bit_data->getscl = i2c_hba_sfp_getscl;
200  bit_data->getsda = i2c_hba_sfp_getsda;
201 
202  if (pdata->udelay)
203  bit_data->udelay = pdata->udelay;
204  else if (pdata->scl_is_output_only)
205  bit_data->udelay = 50; /* 10 kHz */
206  else
207  bit_data->udelay = 5; /* 100 kHz */
208 
209  if (pdata->timeout)
210  bit_data->timeout = pdata->timeout;
211  else
212  bit_data->timeout = HZ / 10; /* 100 ms */
213 
214  //bit_data->data = pdata;
215  bit_data->data = pdev;
216 
217  adap->owner = THIS_MODULE;
218  snprintf(adap->name, sizeof(adap->name), "i2c_hba_sfp%d", pdev->id);
219  adap->algo_data = bit_data;
220  adap->dev.parent = &pdev->dev;
221 
222  ret = i2c_bit_add_bus(adap);
223  if (ret)
224  goto err_add_bus;
225 
226  platform_set_drvdata(pdev, adap);
227 
228  dev_info(&pdev->dev, "using pins %u (SDA) and %u (SCL%s)\n",
229  pdata->sda_pin, pdata->scl_pin,
230  pdata->scl_is_output_only
231  ? ", no clock stretching" : "");
232 
233  return 0;
234 
235 err_add_bus:
236  hba_sfp_free(GETD(pdev), pdata->scl_pin);
237 err_request_scl:
238  hba_sfp_free(GETD(pdev), pdata->sda_pin);
239 err_request_sda:
240  kfree(bit_data);
241 err_alloc_bit_data:
242  kfree(adap);
243 err_alloc_adap:
244  return ret;
245 }

293 static int afhba_sfp_i2c_create1(struct AFHBA_DEV *adev, int ch)
294 {
295  int rc;
296  struct platform_device* pd =
297  kzalloc(2*sizeof(struct platform_device), GFP_KERNEL);
298  pd->name = "i2c_hba_sfp";
299  pd->id = adev->idx*2+CHIX(ch);
300  pd->dev.platform_data = &rtm_t_i2c_gpio_data_templates[CHIX(ch)];
302  pd[1].resource = (struct resource*)adev;
303  rc = platform_device_register(pd);
304 
305  if (rc != 0){
306  dev_err(pdev(adev), "platform device register failed");
307  kfree(pd);
308  }else{
309  adev->hba_sfp_i2c[CHIX(ch)] = pd;
310  }
311  return rc;
312 }
