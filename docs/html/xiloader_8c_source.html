48 static inline u32 bs(u32 num) {
49  u32 swapped = num>>24 & 0x000000ff | // move byte 3 to byte 0
50  num<<8 & 0x00ff0000 | // move byte 1 to byte 2
51  num>>8 & 0x0000ff00 | // move byte 2 to byte 1
52  num<<24 & 0xff000000; // byte 0 to byte 3
53  return swapped;
54 }

278 #define MAX1WRITE 0x100000
279 void load_bitstream(uint32_t* bitstream, int nbytes)
280 /* byte swap it and dump it into the FPGA */
281 {
282  FILE *ofp = stdout;
283  int totwrite = 0;
284  char* bcursor = (char*)bitstream;
285 
286  if (strcmp(OPTS.out, STDOUT_STR) != 0){
287  ofp = fopen (OPTS.out, "w");
288  if (!ofp){
289  perror(OPTS.out);
290  exit(1);
291  }
292  }
293 
294  if (!OPTS.raw){
295  int bit_words = nbytes/sizeof(uint32_t);
296  int ii;
297  for (ii = 0; ii <= bit_words; ii++) {
298  bitstream[ii] = BYTESWAP(bitstream[ii]);
299  }
300  }
301 
302 
303  while (totwrite < nbytes){
304  int req_bytes = nbytes - totwrite;
305  int act_bytes;
306  if (req_bytes > MAX1WRITE) req_bytes = MAX1WRITE;
307  act_bytes = fwrite(bcursor, 1, req_bytes, ofp);
308  if (act_bytes != req_bytes){
309  fprintf(stderr, "ERROR request: %d got %d bytes\n", req_bytes, act_bytes);
310  }
311  bcursor += act_bytes;
312  totwrite += act_bytes;
313  }
314 
315  fclose(ofp);
316 }
