105 static void afhba_spi_start(struct afhba_spi *hw, int on)
106 {
107  struct AFHBA_DEV *adev = hw->adev;
108  if (on){
109  hw->ctrl_mirror |= AFHBA_SPI_CTL_START;
110  }else{
111  hw->ctrl_mirror &= ~AFHBA_SPI_CTL_START;
112  }
113  dev_dbg(pdev(adev), "%08x %s", hw->ctrl_mirror, on? "ON": "OFF");
114  afhba_spi_write_ctl(hw, hw->ctrl_mirror);

116 static inline void afhba_spi_write(struct afhba_spi *hw, unsigned char cc)
117 {
118  struct AFHBA_DEV *adev = hw->adev;
119  dev_dbg(pdev(adev), "\t%04x := %02x", HOST_SPI_FLASH_DATA_REG, cc);
120 
121  afhba_write_reg(hw->adev, HOST_SPI_FLASH_DATA_REG, cc);
122  afhba_spi_start(hw, 1);
123  afhba_spi_wait_busy(hw);
124  afhba_spi_start(hw, 0);
125 }

151 static void rtm_t_spi_chipsel(struct spi_device *spi, int value)
152 {
153  unsigned int cspol = spi->mode & SPI_CS_HIGH ? 1 : 0;
154  struct afhba_spi *hw = to_hw(spi);
155  struct AFHBA_DEV *adev = hw->adev;
156 
157  dev_dbg(pdev(adev), "value %d", value);
158  if (afhba_spi_debug >= 2){
159  afhba_spi_read_ctl(hw);
160  }
161 
162  switch(value){
163  case BITBANG_CS_INACTIVE:
164  afhba_spi_write_ctl(hw, cspol? 0: AFHBA_SPI_CS);
165  break;
166  case BITBANG_CS_ACTIVE:
167  afhba_spi_write_ctl(hw, cspol? AFHBA_SPI_CS: 0);
168  break;
169  }
170 
171  if (afhba_spi_debug >= 2){
172  afhba_spi_read_ctl(hw);
173  }
174 }
175 static int rtm_t_spi_setupxfer(struct spi_device *spi,
176  struct spi_transfer *t)
177 {
178  struct afhba_spi *hw = to_hw(spi);
179 
180  spin_lock(&hw->bitbang.lock);
181  if (!hw->bitbang.busy) {

204 static int rtm_t_spi_txrx(struct spi_device *spi, struct spi_transfer *t)
205 {
206  struct afhba_spi *hw = to_hw(spi);
207  struct AFHBA_DEV *adev = hw->adev;
208 
209  dev_dbg(pdev(adev), "txrx: tx %p, rx %p, len %d",
210  t->tx_buf, t->rx_buf, t->len);
211 
212  hw->tx = t->tx_buf;
213  hw->rx = t->rx_buf;
214 
215  if (hw->tx != NULL && hw->rx != NULL){
216  dev_err(pdev(adev), "tx and rx requested. This is NOT good!");
217  }
218 
219  hw->len = t->len;
220  hw->count = 0;
221 
222  if (hw->tx){
223  /* send the first byte */
224  afhba_spi_write(hw, hw_txbyte(hw, 0));
225 
226  dev_dbg(pdev(adev), "02: done first write now while %d < %d",
227  hw->count, hw->len);
228  ++hw->count;
229  }
230 
231  for (; hw->count < hw->len; ++hw->count){
232  //msleep(spi_write_msleep);
233  yield();
234 
235  if (hw->rx){
236  hw->rx[hw->count] = afhba_spi_read(hw);
237  }
238 
239  if (hw->tx){
240  afhba_spi_write(hw, hw_txbyte(hw, hw->count));
241  }
242  }
243 
244  dev_dbg(pdev(adev), "return %d\n",hw->count);
245  return hw->count;
246 }

272 int afhba_spi_master_init(struct AFHBA_DEV *adev)
273 {
274  struct device *dev = &adev->pci_dev->dev;
275  struct spi_master *master;
276  int rc = 0;
277  struct afhba_spi *hw;
278 
279  dev_info(dev, "call spi_alloc_master, dev %p", dev);
280 
281  master = spi_alloc_master(dev, sizeof(struct afhba_spi));
282  if (master == NULL){
283  dev_err(dev, "No memory for spi_master\n");
284  rc = -ENOMEM;
285  goto err_nomem;
286  }
287  hw = spi_master_get_devdata(master);
288  memset(hw, 0, sizeof(struct afhba_spi));
289 
290  hw->master = spi_master_get(master);
291  hw->master->bus_num = adev->idx;
292 // hw->pdata = pdev->dev.platform_data;
293  hw->dev = dev;
294  hw->adev = adev;
295 
296  hw->bitbang.master = hw->master;
297  hw->bitbang.setup_transfer = rtm_t_spi_setupxfer;
298  hw->bitbang.chipselect = rtm_t_spi_chipsel;
299  hw->bitbang.txrx_bufs = rtm_t_spi_txrx;
300  hw->bitbang.master->setup = rtm_t_spi_setup;
301  hw->bitbang.flags = SPI_CS_HIGH;
302 
303  /* register our spi controller */
304 
305  dev_dbg(dev, "call spi_bitbang_start()");
306  rc = spi_bitbang_start(&hw->bitbang);
307  dev_dbg(dev, "back from spi_bitbang_start()");
308 
309  if (rc) {
310  dev_err(dev, "Failed to register SPI master\n");
311  goto err_register;
312  }
313 
314  dev_dbg(pdev(adev), "registering %s\n", rtm_t_spi_devices[0].modalias);
315  spi_new_device(master, &rtm_t_spi_devices[0]);
316 
317  return 0;
318 
319 err_register:
320 
321 err_nomem:
322  return rc;
323 }
